def audit_network(self):
    try:
        # Получаем список всех слушающих TCP/UDP портов
        out = subprocess.run(['ss', '-tulpn'], capture_output=True, text=True, timeout=30).stdout
        
        # Список известных опасных служб для формирования описания
        dangerous = {"21": "FTP", "23": "Telnet", "25": "SMTP", "3306": "MySQL", "5432": "PostgreSQL"}
        my_web_port = "5000" # Порт вашей веб-оболочки

        for line in out.split('\n'):
            if "LISTEN" in line or "UNCONN" in line:
                # Извлекаем порт из строки (обычно в формате *:порт или 0.0.0.0:порт)
                import re
                port_match = re.search(r':(\d+)\s', line)
                if port_match:
                    port = port_match.group(1)
                    
                    # ПРОПУСКАЕМ порт веб-оболочки, чтобы скрипт не упал
                    if port == my_web_port:
                        continue
                    
                    desc = dangerous.get(port, "Unknown Service")
                    
                    # Логика поиска процесса (из вашего кода)
                    proc = ""
                    m = re.search(r'users:\(\("([^"]+)"', line)
                    if m: proc = m.group(1)
                    
                    # Формируем команду исправления
                    fix = f"sudo fuser -k {port}/tcp && sudo systemctl stop {proc if proc else desc.lower()} 2>/dev/null"
                    
                    self.add("Сетевой аудит", f"Открыт порт {port} ({desc})" + (f" - процесс: {proc}" if proc else ""), fix)
    except Exception as e:
        print(f"Ошибка аудита сети: {e}")
        pass
